CREATE TABLE investment_participants (
    id UUID PRIMARY KEY DEFAULT uuid_generate_v4(),
    investment_pool_id UUID REFERENCES investment_pools(id),
    member_id UUID REFERENCES apartment_members(id),
    amount DECIMAL(10,2),
    ownership_percentage DECIMAL(5,2),
    joined_at TIMESTAMPTZ DEFAULT NOW()
);

-- Developer Donations
CREATE TABLE developer_donations (
    id UUID PRIMARY KEY DEFAULT uuid_generate_v4(),
    user_id UUID REFERENCES auth.users(id),
    apartment_id UUID REFERENCES apartments(id),
    amount DECIMAL(10,2),
    donation_type VARCHAR(20),
    ad_count INTEGER,
    donated_at TIMESTAMPTZ DEFAULT NOW()
);

-- Guests
CREATE TABLE guests (
    id UUID PRIMARY KEY DEFAULT uuid_generate_v4(),
    apartment_id UUID REFERENCES apartments(id),
    sponsor_member_id UUID REFERENCES apartment_members(id),
    name VARCHAR(255),
    relationship VARCHAR(100),
    phone VARCHAR(20),
    email VARCHAR(255),
    photo_url TEXT,
    id_proof_url TEXT,
    stay_from DATE,
    stay_to DATE,
    status VARCHAR(20) DEFAULT 'pending',
    entry_code VARCHAR(20),
    created_at TIMESTAMPTZ DEFAULT NOW()
);
